class AdminUser {
  // بيانات المسؤول
  - id, email, fullName, role, phoneNumber
  - profileImage, isActive, permissions
  - toFirestore(), fromFirestore()
}

class DoctorRequest {
  // بيانات طلب الطبيب
  - doctorId, fullName, specialty
  - medicalLicense, medicalDegree
  - clinicName, clinicAddress, documents
  - status (pending/approved/rejected)
  - toFirestore(), fromFirestore(), copyWith()
}

class AdminStats {
  // إحصائيات الإدارة
  - totalDoctors, pendingRequests
  - approvedDoctors, totalPatients
  - totalAppointments, averageDoctorRating
}

## 🔄 دورة حياة الطبيب

```
1. الطبيب يتقدم بطلب
   └── status: pending

2. مسؤول يراجع الطلب
   ├── يوافق → status: approved
   └── يرفض → status: rejected

3. تُسجل العملية في السجل
   └── admin_logs

4. يُعدل حساب الطبيب
   ├── isVerified: true (if approved)
   └── حساب معطل (if rejected)
```
